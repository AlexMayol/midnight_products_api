CREATE TABLE products(
    id        serial PRIMARY KEY,
    category serial references categories(id),
    name       char(150) NULL,
    en_description       char(500) NULL,
    es_description       char(500) NULL,
    de_description       char(500) NULL,
    price       decimal(10) NULL,
    image       char(300) NULL
);

CREATE TABLE categories(
    id        serial PRIMARY KEY,
    name       char(50) NULL,
    description       char(150) NULL,
    image       char(200) NULL
);


CREATE TABLE colors (
    id        serial PRIMARY KEY,
    code       char(20) NULL,
    label       char(50) NULL
);

CREATE TABLE product_colors(
    id        serial PRIMARY KEY,
    product serial references products(id),
    color serial references colors(id)
);

    id        serial PRIMARY KEY,
